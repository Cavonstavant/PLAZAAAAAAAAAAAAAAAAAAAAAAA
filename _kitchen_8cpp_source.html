98 void Kitchen::_receptCook(Kitchen *obj)
99 {
100  while (true) {
101  std::this_thread::sleep_for(std::chrono::milliseconds(10));
102  std::string fullCommand = obj->commandQueue.get()->receiveMessage();
103 
104  if (fullCommand.find("avail_slots:") == 0) {
105  obj->commandQueue.get()->sendMessage(fullCommand);
106  continue;
107  }
108  if (_isAvailableCook(fullCommand)) {
109  std::this_thread::sleep_for(std::chrono::milliseconds(10));
110  std::string message = "avail_slots:" + std::to_string(obj->_orders.size() + (obj->_nbCooks - obj->_availCooks)) + ";fridge:";
111  {
112  std::unique_lock<std::mutex> lock(obj->_fridgeMutex);
113  for (size_t x = 1; x < IngredientNumber; x++) {
114  message.append(obj->_fridge[x].name);
115  message.append("," + std::to_string(obj->_fridge[x].number) + ";");
116  }

118  obj->commandQueue.get()->sendMessage(message);

120  }

146 void Kitchen::_Cook(Kitchen *obj)
147 {
148  Cook cook;
149  bool asEnoughIngredients = false;
150  cook.setCookingTimeMultipiler(obj->_cookingTime);
151 
152  while (true) {
153  Pizza order;
154  {
155  std::unique_lock<std::mutex> lock(obj->_mutex);
156  obj->order_condition.wait(lock, [&obj] {
157  return !obj->_orders.empty() || obj->_stopKitchen;
158  });
159  if (obj->_stopKitchen) {
160  return;
161  }
162  order = obj->_orders.front();
163  obj->_orders.pop();
164  obj->_availCooks--;
165  }
166  {
167  std::unique_lock<std::mutex> lock(obj->_fridgeMutex);
168  while (!asEnoughIngredients) {
169  _waitToFillFridge(obj->_refillTime, obj);
170  try {
171  obj->_fridge -= order;
172  } catch (...) {
173  continue;
174  }
175  asEnoughIngredients = true;
176  }
177  }
178  cook.cookPizza(order);
179  {
180  std::unique_lock<std::mutex> lock(obj->_mutex);
181  obj->_availCooks++;
182  }
183  }
184 }
