118 void Reception::_manageOrders(const InputParser &server)
119 {
120  {
121  std::unique_lock<std::mutex> lock(_mutex);
122  std::vector<Pizza> orders = server.getPizza();
123  std::size_t nbKitchen = _kitchenMap.size();
124  std::size_t nbMaxKitchen = 9;
125  std::size_t totalAvail = 0;
126  std::string message;
127  std::vector<std::size_t> availSlots;
128 
129  for (Pizza order: orders) {
130  availSlots.clear();
131  totalAvail = 0;
132  nbKitchen = _kitchenMap.size();
133  for (auto it: _kitchenMap) {
134  it.second.get()->sendMessage("avail_slots?");
135  std::this_thread::sleep_for(std::chrono::milliseconds(10));
136  message = it.second.get()->receiveMessage();
137  while (_isAvailableSlots(message) < 0) {
138  it.second.get()->sendMessage(message);
139  std::this_thread::sleep_for(std::chrono::milliseconds(100));
140  message = it.second.get()->receiveMessage();
141  }
142  availSlots.push_back(_cookNumber * 2 - _isAvailableSlots(message));
143  }
144  for (std::size_t it: availSlots)
145  totalAvail += it;
146  if (order.number > totalAvail) {
147  if (nbKitchen < nbMaxKitchen) {
148  std::size_t nbNewKitchen = (order.number % (_cookNumber * 2) == 0) ? order.number / (_cookNumber * 2) : order.number / (_cookNumber * 2) + 1;
149  if (nbNewKitchen + nbKitchen > nbMaxKitchen) {
150  nbNewKitchen = nbMaxKitchen - nbKitchen;
151  }
152  pid_t newKitchenPid;
153  for (; nbNewKitchen > 0; nbNewKitchen--) {
154  Kitchen newKitchen(_cookNumber, _refillTime, _cookingTime);
155  std::shared_ptr<MessageQueue> newQueue = std::make_shared<MessageQueue>();
156  newQueue.get()->openQueue(std::string("/plazzaQueueNumber" + std::to_string(_kitchenMap.size())));
157  newKitchenPid = fork();
158  if (newKitchenPid == 0) {
159  newKitchen.commandQueue = newQueue;
160  newKitchen.start();
161  newKitchen.work();
162  exit(0);
163  } else {
164  _kitchenMap[newKitchenPid] = newQueue;
165  availSlots.push_back(_cookNumber * 2);
166  }
167  }
168  }
169  }
170  for (; order.number > 0; order.number--) {
171  std::size_t indexMin = 0;
172  std::size_t index = 0;
173  Pizza tmp = order;
174  tmp.number = 1;
175  for (std::size_t it: availSlots) {
176  if (it < availSlots[indexMin])
177  indexMin = index;
178  index++;
179  }
180  index = 0;
181  for (auto it: _kitchenMap) {
182  if (index == indexMin)
183  it.second.get()->sendMessage(pack(tmp));
184  index++;
185  }
186  }
187  }
188  }
189 }
