48 void Reception::_sendCommand(const InputParser &command)
49 {
50  std::vector<Pizza> pizzaToCook = command.getPizza();
51 
52  if (pizzaToCook.size() == 0)
53  return;
54  if (_kitchenMap.size() == 9) {
55  MessageQueueEx("You can not open more than 9 Kitchens", Logger::MEDIUM);
56  return;
57  }
58 
59  Kitchen newKitchen(_cookNumber, _refillTime, _cookingTime);
60  std::shared_ptr<MessageQueue> newQueue = std::make_shared<MessageQueue>();
61 
62  newQueue.get()->openQueue(std::string("/plazzaQueueNumber" + std::to_string(_kitchenMap.size())));
63 
64  pid_t newKitchenPid = fork();
65 
66  if (newKitchenPid == 0) {
67  newKitchen.commandQueue = newQueue;
68  newKitchen.start();
69  newKitchen.stop();
70  } else {
71  _kitchenMap[newKitchenPid] = newQueue;
72 
73  for (std::size_t x = 0; x < pizzaToCook.size(); ++x) {
74  std::this_thread::sleep_for(std::chrono::milliseconds(10));
75  newQueue.get()->sendMessage(pack(pizzaToCook[x]));
76  }
77  }
78 }
